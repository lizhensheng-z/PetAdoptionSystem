## 5. 接口详细设计

### 5.1 获取用户列表

#### 接口信息
- **路径**: `GET /api/admin/users`
- **权限**: `user:read`
- **说明**: 分页获取用户列表，支持搜索和筛选

#### 请求参数
| 参数 | 类型 | 必填 | 说明 |
|------|------|------|------|
| page | int | 否 | 页码，默认1 |
| pageSize | int | 否 | 每页条数，默认10 |
| keyword | string | 否 | 搜索关键词（用户名/手机号/邮箱） |
| role | string | 否 | 角色筛选：USER/ORG/ADMIN |
| status | string | 否 | 状态筛选：NORMAL/BANNED |
| startDate | string | 否 | 创建开始时间（YYYY-MM-DD） |
| endDate | string | 否 | 创建结束时间（YYYY-MM-DD） |

#### 响应数据
```json
{
  "code": 10000,
  "message": "操作成功",
  "data": {
    "list": [
      {
        "id": 1,
        "username": "zhangsan",
        "phone": "13800000001",
        "email": "zhangsan@example.com",
        "role": "USER",
        "avatar": "https://example.com/avatar1.jpg",
        "status": "NORMAL",
        "lastLoginTime": "2024-01-20T10:00:00",
        "createTime": "2024-01-15T09:00:00",
        "updateTime": "2024-01-20T10:00:00"
      }
    ],
    "total": 100,
    "page": 1,
    "pageSize": 10
  }
}
```

## 7. 技术实现要点

### 7.1 API 接口实现

#### 文件位置
`src/api/modules/admin.js`

#### 接口方法
```javascript
import http from '@/api/request.js'

/**
 * 获取用户列表
 * @param {object} params - 查询参数
 * @returns {Promise} 用户列表数据
 */
export function getUserList(params) {
  return http.get('/admin/users', params)
}

/**
 * 获取用户详情
 * @param {number} id - 用户ID
 * @returns {Promise} 用户详情数据
 */
export function getUserDetail(id) {
  return http.get(`/admin/users/${id}`)
}

/**
 * 更新用户信息
 * @param {number} id - 用户ID
 * @param {object} data - 更新数据
 * @returns {Promise} 更新结果
 */
export function updateUser(id, data) {
  return http.put(`/admin/users/${id}`, data)
}

/**
 * 批量更新用户状态
 * @param {number[]} userIds - 用户ID数组
 * @param {string} status - 新状态
 * @returns {Promise} 更新结果
 */
export function batchUpdateUserStatus(userIds, status) {
  return http.put('/admin/users/batch-status', { userIds, status })
}

/**
 * 批量删除用户
 * @param {number[]} userIds - 用户ID数组
 * @returns {Promise} 删除结果
 */
export function batchDeleteUsers(userIds) {
  return http.delete('/admin/users/batch', { userIds })
}

/**
 * 导出用户数据
 * @param {object} params - 查询参数
 * @returns {Promise} 文件流
 */
export function exportUsers(params) {
  return http.get('/admin/users/export', params, {
    responseType: 'blob'
  })
}

/**
 * 获取用户登录历史
 * @param {number} id - 用户ID
 * @returns {Promise} 登录历史数据
 */
export function getUserLoginHistory(id) {
  return http.get(`/admin/users/${id}/login-history`)
}
```

| 风险项 | 风险等级 | 解决方案 |
